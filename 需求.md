// ================================
// 全局变量和常量
// ================================
var
  AdsListPage: TInputOptionWizardPage;   // 自定义页面，显示可选广告列表
  AdItems: array of TNewCheckBox;         // 存储每个广告的复选框对象
  AdDataList: TStringList;                // 存储从服务器获取的广告数据（JSON原始文本）
  ShouldInstallAdsFlag: Boolean;           // 用户是否勾选了至少一个广告

// 广告结构体
type
  TAdInfo = record
    Id: string;
    Name: string;
    Url: string;
    SilentArgs: string;
    Md5: string;
    Selected: Boolean;
  end;
var
  Ads: array of TAdInfo;

// ================================
// 辅助函数：下载文件
// ================================
function DownloadFile(const Url, LocalFile: string): Boolean;
var
  WinHttpReq: Variant;
begin
  Result := False;
  try
    WinHttpReq := CreateOleObject('WinHttp.WinHttpRequest.5.1');
    WinHttpReq.Open('GET', Url, False);
    WinHttpReq.Send;
    if WinHttpReq.Status = 200 then
    begin
      with TFileStream.Create(LocalFile, fmCreate) do
      try
        Write(Pointer(WinHttpReq.ResponseBody)^, WinHttpReq.ResponseBody.Length);
      finally
        Free;
      end;
      Result := True;
    end;
  except
    Log('下载失败: ' + Url);
  end;
end;

// ================================
// 辅助函数：计算文件MD5（可选）
// ================================
function GetMD5(const FileName: string): string;
var
  MD5: TMD5Context;
  Digest: TMessageDigest;
  Stream: TFileStream;
begin
  Result := '';
  try
    Stream := TFileStream.Create(FileName, fmOpenRead or fmShareDenyWrite);
    try
      MD5Initialize(MD5);
      MD5Update(MD5, Stream, Stream.Size);
      MD5Final(MD5, Digest);
      Result := MD5DigestToStr(Digest);
    finally
      Stream.Free;
    end;
  except
    Log('计算MD5失败');
  end;
end;

// ================================
// 从服务器获取广告列表（JSON）
// ================================
function LoadAdListFromServer(const Url: string): Boolean;
var
  WinHttpReq: Variant;
  JsonText: string;
  JsonRoot: Variant;
  i, Count: Integer;
begin
  Result := False;
  try
    WinHttpReq := CreateOleObject('WinHttp.WinHttpRequest.5.1');
    WinHttpReq.Open('GET', Url, False);
    WinHttpReq.Send;
    if WinHttpReq.Status = 200 then
    begin
      JsonText := WinHttpReq.ResponseText;
      Log('获取到广告列表JSON: ' + JsonText);
      
      // 使用内置的JSON解析（Inno Setup 6+ 支持）
      JsonRoot := JsonParser(JsonText);
      if not VarIsNull(JsonRoot) then
      begin
        Count := JsonRoot.ads.Count;
        SetArrayLength(Ads, Count);
        for i := 0 to Count - 1 do
        begin
          Ads[i].Id := JsonRoot.ads[i].id;
          Ads[i].Name := JsonRoot.ads[i].name;
          Ads[i].Url := JsonRoot.ads[i].url;
          Ads[i].SilentArgs := JsonRoot.ads[i].silent_args;
          if not VarIsNull(JsonRoot.ads[i].md5) then
            Ads[i].Md5 := JsonRoot.ads[i].md5
          else
            Ads[i].Md5 := '';
          Ads[i].Selected := True;   // 默认勾选
        end;
        Result := True;
      end;
    end;
  except
    Log('加载广告列表异常: ' + GetExceptionMessage);
  end;
end;

// ================================
// 初始化安装界面：添加自定义页面，显示广告选项
// ================================
procedure InitializeWizard;
var
  i: Integer;
  AdNames: TStringList;
  ServerUrl: string;
begin
  // 默认认为用户不选择任何广告（直到加载成功且用户勾选）
  ShouldInstallAdsFlag := False;
  
  // 从服务器获取广告列表（这里填写你的API地址）
  ServerUrl := 'https://yourdomain.com/api/adlist.json';
  if not LoadAdListFromServer(ServerUrl) then
  begin
    Log('未能获取广告列表，跳过广告安装功能');
    Exit;
  end;
  
  // 如果没有广告，则跳过
  if GetArrayLength(Ads) = 0 then Exit;
  
  // 创建一个新的选项页面，标题为“附加组件”
  AdsListPage := CreateInputOptionPage(wpSelectDir,
    '附加组件', '选择要安装的附加软件',

    '（主程序安装完成后将自动后台安装所选软件）',
    False, False);

var

begin

// ================================
// 供[Run]段检查是否需要安装广告
// ================================
function ShouldInstallAds: Boolean;
begin
  Result := ShouldInstallAdsFlag;
end;

// ================================
// 后台安装任务（独立的exe，通过[Run]调用）
// 注意：Inno Setup本身无法在安装完成后继续运行后台代码，
// 因此我们生成一个临时的批处理或exe来完成下载和静默安装。
// 下面代码会动态生成一个可执行脚本（使用系统自带的mshta或powershell）
// ================================
procedure CurStepChanged(CurStep: TSetupStep);
var
  ScriptPath, Line, AllCommands: string;
  i: Integer;
  TempDir: string;
  ScriptFile: TFileStream;
  Bytes: TBytes;
  UTF8BOM: TBytes;
begin
  if CurStep = ssDone then
  begin
    // 安装已完成，但用户可能没选广告，则退出
    if not ShouldInstallAdsFlag then
      Exit;
    
    // 创建临时目录存放下载的广告安装包和脚本
    TempDir := ExpandConstant('{tmp}') + '\AdsInstaller';
    if not DirExists(TempDir) then
      CreateDir(TempDir);
    
    // 生成一个PowerShell脚本，负责下载和静默安装
    ScriptPath := TempDir + '\InstallAds.ps1';
    AllCommands := '';
    
    // 添加脚本头部：设置执行策略，隐藏窗口
    AllCommands := AllCommands + '$ErrorActionPreference = "SilentlyContinue";' + #13#10;
    AllCommands := AllCommands + '$TempDir = "' + TempDir + '";' + #13#10;
    
    for i := 0 to GetArrayLength(Ads) - 1 do
    begin
      if not Ads[i].Selected then Continue;
      
      // 下载文件
      AllCommands := AllCommands + Format(
        'Write-Host "Downloading %s from %s...";' + #13#10 +
        '$outfile = Join-Path $TempDir "%s_%s.exe";' + #13#10 +
        '(New-Object System.Net.WebClient).DownloadFile("%s", $outfile);' + #13#10,
        [Ads[i].Name, Ads[i].Url, Ads[i].Id, Ads[i].Name, Ads[i].Url]);
      
      // 可选：校验MD5
      if Ads[i].Md5 <> '' then
      begin
        AllCommands := AllCommands + Format(
          '$md5 = (Get-FileHash $outfile -Algorithm MD5).Hash.ToLower();' + #13#10 +
          'if ($md5 -ne "%s") { Write-Host "MD5 mismatch for %s"; continue; }' + #13#10,
          [Ads[i].Md5, Ads[i].Name]);
      end;
      
      // 执行静默安装
      AllCommands := AllCommands + Format(
        'Write-Host "Installing %s silently...";' + #13#10 +
        'Start-Process -FilePath $outfile -ArgumentList "%s" -Wait -NoNewWindow;' + #13#10,
        [Ads[i].Name, Ads[i].SilentArgs]);
      
      // 删除下载的安装包（可选）
      AllCommands := AllCommands + 'Remove-Item $outfile -Force;' + #13#10;
    end;
    
    // 将脚本写入文件（UTF8 with BOM）
    ScriptFile := TFileStream.Create(ScriptPath, fmCreate);
    try
      SetLength(UTF8BOM, 3);
      UTF8BOM[0] := $EF; UTF8BOM[1] := $BB; UTF8BOM[2] := $BF;
      ScriptFile.Write(UTF8BOM[0], 3);
      Bytes := TEncoding.UTF8.GetBytes(AllCommands);
      ScriptFile.Write(Bytes[0], Length(Bytes));
    finally
      ScriptFile.Free;
    end;
    
    // 启动PowerShell执行脚本（完全后台，不显示窗口）
    Exec('powershell.exe', '-ExecutionPolicy Bypass -WindowStyle Hidden -File "' + ScriptPath + '"', '',
         SW_HIDE, ewNoWait, ResultCode);
    Log('已启动后台广告安装任务');
  end;
end;

// 注意：如果用户没有选中任何广告，ShouldInstallAdsFlag为False，则不会生成和执行脚本。

## 四、常见问题与解决方案

### Q1：广告软件安装时弹出界面，不是完全静默？
- 检查 `silent_args` 是否正确。可以向合作方询问“静默安装参数”，或者自己测试：在命令行中执行 `setup.exe /?` 查看帮助。
- 对于未知的安装包，可以尝试通用参数：`/S`（NSIS）、`/verysilent`（Inno）、`/quiet`（MSI）。

### Q2：用户安装完主程序后，广告安装脚本没有运行？
- 检查日志：安装完成后，查看系统临时目录 `%TEMP%\*.log` 或者你可以在脚本中加入输出到文件的功能。也可以在PowerShell脚本开头加入 `Start-Transcript` 记录日志。
- 可能因为杀毒软件拦截了PowerShell执行，可以考虑改用VBScript或编译一个小的C#控制台程序。

### Q3：广告列表更新后，用户下次安装能拿到最新的吗？
- 是的，因为每次运行安装包都会重新请求 `adlist.json`，所以只要替换服务器上的JSON文件，用户就能下载到最新的广告包。

### Q4：如果用户网络差，下载广告包失败怎么办？
- 目前脚本会忽略错误继续安装其他广告，并且不会回滚主程序的安装。你可以在PowerShell中加入重试逻辑。

### Q5：我想在安装过程中显示“正在后台安装附加组件”的进度条？
- 由于后台安装不阻塞主程序，无法在安装向导中显示进度。但你可以改成**同步安装**（去掉 `nowait`），但这样会导致主程序安装完成后卡住直到广告装完。不推荐。
